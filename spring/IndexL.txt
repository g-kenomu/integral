    <script>
        let projects = JSON.parse(localStorage.getItem('dev-empire-projects')) || [];
        const targetAmount = 100000;

        function save() {
            localStorage.setItem('dev-empire-projects', JSON.stringify(projects));
            render();
        }

        function addProject() {
            const input = document.getElementById('new-idea');
            if (!input.value.trim()) return;
            projects.unshift({
                id: Date.now(),
                title: input.value,
                status: 'planned',
                potential: Math.floor(Math.random() * 5000) + 1000
            });
            input.value = '';
            save();
        }

        function deleteProject(id) {
            projects = projects.filter(p => p.id !== id);
            save();
        }

        function updateStatus(id, status) {
            const p = projects.find(p => p.id === id);
            if (p) p.status = status;
            save();
        }

        function generateIdea() {
            const targets = ['学生', '社会人', 'ゲーマー', '犬好き', 'ダイエット中'];
            const types = ['タイマー', 'ToDo', '診断アプリ', '家計簿', 'AIチャット'];
            const idea = `${targets[Math.floor(Math.random()*targets.length)]}向け${types[Math.floor(Math.random()*types.length)]}`;
            document.getElementById('new-idea').value = idea;
        }

        function render() {
            const list = document.getElementById('project-list');
            list.innerHTML = '';
            
            let shippedCount = 0;
            let revenue = 0;

            projects.forEach(p => {
                if (p.status === 'shipped') {
                    shippedCount++;
                    revenue += p.potential;
                }

                const card = document.createElement('div');
                card.className = `p-4 rounded-xl border ${p.status === 'shipped' ? 'bg-slate-900 border-emerald-900/50 opacity-75' : 'bg-slate-800 border-slate-700'}`;
                card.innerHTML = `
                    <div class="flex justify-between items-start">
                        <h3 class="font-bold text-lg ${p.status === 'shipped' ? 'text-emerald-400 line-through' : ''}">${p.title}</h3>
                        <button onclick="deleteProject(${p.id})" class="text-slate-600 hover:text-red-400"><i data-lucide="trash-2" class="w-4 h-4"></i></button>
                    </div>
                    <div class="flex justify-between items-center mt-4">
                        <div class="flex gap-2">
                            ${['planned', 'building', 'shipped'].map(s => `
                                <button onclick="updateStatus(${p.id}, '${s}')" class="px-2 py-1 rounded text-[10px] font-bold uppercase ${p.status === s ? 'bg-emerald-500 text-slate-900' : 'bg-slate-700 text-slate-400'}">
                                    ${s === 'planned' ? '構想' : s === 'building' ? '開発' : '完了'}
                                </button>
                            `).join('')}
                        </div>
                        ${p.status === 'shipped' ? `<span class="text-emerald-500 text-sm font-bold">¥${p.potential}</span>` : ''}
                    </div>
                `;
                list.appendChild(card);
            });

            document.getElementById('shipped-count').innerHTML = `${shippedCount}<span class="text-sm font-normal text-slate-500">本</span>`;
            document.getElementById('total-revenue').innerText = `¥${revenue.toLocaleString()}`;
            const percent = Math.min((revenue / targetAmount) * 100, 100);
            document.getElementById('progress-text').innerText = `${percent.toFixed(1)}%`;
            document.getElementById('progress-bar').style.width = `${percent}%`;
            
            lucide.createIcons();
        }

        render();
    </script>
